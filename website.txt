(function(){
  var hamburger = document.getElementById('hamburger');
  var navLinks = document.getElementById('navLinks');
  if(hamburger && navLinks){
    hamburger.addEventListener('click', function(){
      var isOpen = navLinks.classList.toggle('open');
      hamburger.setAttribute('aria-expanded', isOpen ? 'true' : 'false');
    });
    navLinks.querySelectorAll('a').forEach(function(a){
      a.addEventListener('click', function(){
        navLinks.classList.remove('open');
        hamburger.setAttribute('aria-expanded', 'false');
      });
    });
  }

  function updateOpenStatus(){
    var el = document.getElementById('openStatus');
    var dot = document.getElementById('openDot');
    if(!el || !dot) return;
    var now = new Date();
    var h = now.getHours() + now.getMinutes() / 60;
    var isOpen = h >= 8 && h < 20;
    el.textContent = isOpen ? 'Otevřeno · dnes do 20:00' : 'Zavřeno · otevíráme od 8:00';
    dot.classList.toggle('is-open', isOpen);
    dot.classList.toggle('is-closed', !isOpen);
  }
  updateOpenStatus();
  setInterval(updateOpenStatus, 60000);

  var reduceMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;
  var reveals = document.querySelectorAll('.reveal');
  if(reduceMotion || !('IntersectionObserver' in window)){
    reveals.forEach(function(el){ el.classList.add('in-view'); });
  } else {
    var io = new IntersectionObserver(function(entries){
      entries.forEach(function(entry){
        if(entry.isIntersecting){
          entry.target.classList.add('in-view');
          io.unobserve(entry.target);
        }
      });
    }, {threshold:0.15, rootMargin:'0px 0px -40px 0px'});
    reveals.forEach(function(el){ io.observe(el); });
  }
})();
